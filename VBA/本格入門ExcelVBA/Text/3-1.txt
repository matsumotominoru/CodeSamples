
'---------------------------------------------
'事例1
'保存場所を特定してブックを開く
'
'(ドライブ名を任意に変更してから実行してください)
'---------------------------------------------

Sub ブックを開く()
    Workbooks.Open Filename:="C:\honkaku\Dummy.xlsx"
End Sub

'----------------------------------------------------------------------------------------------
'事例2
'カレントフォルダのブックを開く
'
'(「Dummy.xlsx」を閉じて、「honkaku」フォルダをカレントフォルダにしてから実行してください)
'----------------------------------------------------------------------------------------------

Sub ブックを開く2()
    Workbooks.Open Filename:="Dummy.xlsx"
End Sub

'------------------------------------------
'事例3
'ブックを保存して閉じる
'
'(「Dummy.xlsx」を開いてから実行してください)
'------------------------------------------

Sub ブックを閉じる()
    Workbooks("Dummy.xlsx").Close SaveChanges:=True     '名前付き引数
End Sub
